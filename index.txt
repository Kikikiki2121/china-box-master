    <script type="importmap">
        {
            "imports": {
                "three": "https://unpkg.com/three@0.160.0/build/three.module.js",
                "three/addons/": "https://unpkg.com/three@0.160.0/examples/jsm/"
            }
        }
    </script>

    <script type="module">
        import * as THREE from 'three';
        import { DeviceOrientationControls } from 'three/addons/controls/DeviceOrientationControls.js';

        // --- ЛОГГЕР ---
        const logBox = document.getElementById('console-log');
        function log(msg) {
            const line = document.createElement('div');
            line.textContent = `> ${msg}`;
            logBox.prepend(line);
            console.log(msg);
        }

        let camera, scene, renderer, controls;
        const btnStart = document.getElementById('btn-start');
        const overlay = document.getElementById('overlay');

        // 1. Создаем сцену сразу, но не запускаем анимацию
        scene = new THREE.Scene();
        camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        // Отодвигаем камеру чуть назад, чтобы видеть оси, если что
        camera.position.set(0, 0, 0.1); 

        renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.body.appendChild(renderer.domElement);

        // 2. Добавляем "Оси координат" (Helper)
        // Красная = X, Зеленая = Y, Синяя = Z. Если они крутятся - гироскоп работает!
        const axesHelper = new THREE.AxesHelper( 5 );
        scene.add( axesHelper );

        // 3. Добавляем Сферу (Звезды)
        const geometry = new THREE.SphereGeometry(500, 32, 16);
        // Выворачиваем наизнанку
        geometry.scale(-1, 1, 1);
        
        // Загрузка текстуры
        log("Загрузка текстуры...");
        const loader = new THREE.TextureLoader();
        // Используем очень надежную ссылку или цветную заглушку
        loader.load(
            'https://upload.wikimedia.org/wikipedia/commons/c/c3/NGC_4414_%28NASA-med%29.jpg', 
            (texture) => {
                log("Текстура загружена!");
                const material = new THREE.MeshBasicMaterial({ map: texture });
                const sphere = new THREE.Mesh(geometry, material);
                scene.add(sphere);
            },
            undefined,
            (err) => {
                log("ОШИБКА ТЕКСТУРЫ. Включаю каркас.");
                // Если картинка не загрузилась, покажем розовую сетку
                const material = new THREE.MeshBasicMaterial({ color: 0xff00ff, wireframe: true });
                const sphere = new THREE.Mesh(geometry, material);
                scene.add(sphere);
            }
        );

        // 4. Логика кнопки (Самое важное для iPhone)
        btnStart.addEventListener('click', async () => {
            log("Нажата кнопка старта...");
            overlay.style.display = 'none';

            if (window.DeviceOrientationEvent && typeof DeviceOrientationEvent.requestPermission === 'function') {
                log("Это iPhone (iOS 13+). Запрашиваю права...");
                try {
                    const response = await DeviceOrientationEvent.requestPermission();
                    log("Ответ iOS: " + response);
                    
                    if (response === 'granted') {
                        initGyro();
                    } else {
                        log("ОТКАЗАНО В ДОСТУПЕ. Перезагрузи сайт.");
                        alert("Вы запретили доступ. Без этого не будет работать.");
                    }
                } catch (e) {
                    log("Ошибка iOS: " + e.message);
                }
            } else {
                log("Это Android или ПК. Пробуем подключиться...");
                initGyro();
            }
        });

        function initGyro() {
            try {
                controls = new DeviceOrientationControls(camera);
                log("Контроллер создан. Вращай телефон!");
                animate();
            } catch (e) {
                log("Ошибка создания контроллера: " + e.message);
            }
        }

        function animate() {
            requestAnimationFrame(animate);
            if (controls) {
                controls.update();
                // Выводим данные гироскопа в лог (для проверки)
                // log("Alpha: " + controls.deviceOrientation.alpha.toFixed(1)); 
            }
            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

    </script>
